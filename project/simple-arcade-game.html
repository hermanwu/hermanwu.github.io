In this project, I practised JavaScript object oriented programming features to write eloquently designed classes capable of creating countless instances of similarly functioning objects.

Tried a variety of ways that inheritance and delegation can be used to create well-architected and performant applications.

Code Snippet

"use strict";
// Entity represents a single element in the game
var Entity = function(){
    //width and height are shared by all entities
    //which are used to calcuate the relative positions
    this.width = 101;
    this.height = 83;
    //default relative position is 0;
    this.x = 0;
    this.y = 0;
};

//Player which can be controlled using keyboard
var Player = function(){
    Entity.call(this);
    //Player's character image
    this.sprite = 'images/char-boy.png';
    //Starting position on the game board
    this.startingRow = 5;
    this.startingCol = 2;
    //Current position on the game board
    this.row = this.startingRow;
    this.col = this.startingCol;
    //Indicates how many gems have been collected by user
    this.collectedGems = 0;
    //This flag indicate whether player can move or not
    this.moveAbility = true;
};

// Inherit the Entity object;
Player.prototype = Object.create(Entity.prototype);
Player.prototype.contructor = Player;

//Method to draw player on the game board
Player.prototype.render = function(){
    ctx.drawImage(Resources.get(this.sprite), this.x, this.y);
};
